1. 单节点查找
MATCH (l:Law {name: '中华人民共和国招标投标法'}) RETURN l

MATCH (a:Article {article_number: '22'}) RETURN a

MATCH (c:Case {case_id: 'case_001'}) RETURN c

2. 1-hop 关联
MATCH (l:Law)-[:HAS_ARTICLE]->(a:Article) WHERE l.name = '中华人民共和国招标投标法' RETURN l, a

MATCH (c:Case)-[:CITED_IN]->(a:Article) WHERE c.case_id = 'case_001' RETURN c, a

MATCH (sub:Law)-[:IMPLEMENTS]->(main:Law) WHERE sub.name = '招标投标法实施条例' RETURN sub, main

3. 2-hop 探索（扇出分析）
MATCH (l:Law)-[:HAS_ARTICLE]->(a:Article)<-[:CITED_IN]-(c:Case)
WHERE l.name = '中华人民共和国招标投标法'
RETURN l, a, c

MATCH (l1:Law)-[:OVERLAPS_WITH]-(l2:Law)-[:IMPLEMENTS]-(sub:Law)
WHERE l1.name = '中华人民共和国政府采购法'
RETURN l1, l2, sub

4. 变长路径（2 到 4 跳）
MATCH path = (start:Law {name: '广东省实施<政府采购法>办法'})-[:IMPLEMENTS*1..4]->(end:Law)
RETURN path

MATCH path = (l:Law)-[:OVERLAPS_WITH*1..3]-(other:Law)
RETURN path

5. 条件过滤 + 聚合
MATCH (l:Law)-[:HAS_ARTICLE]->(a:Article)
RETURN l.name AS LawName, count(a) AS ArticleCount
ORDER BY ArticleCount DESC

MATCH (c:Case)-[:CITED_IN]->(a:Article)
WITH c, count(a) as cited_count
WHERE cited_count > 1
RETURN c.title, cited_count

MATCH (l:Law)
RETURN l.level, count(l) as total

6. WITH 管道 + 排序
MATCH (sub:Law)-[:IMPLEMENTS]->(main:Law)
WITH main, count(sub) as sub_count
ORDER BY sub_count DESC
LIMIT 1
RETURN main.name, sub_count

MATCH (c:Case)-[:CITED_IN]->(a:Article)
WITH c, count(a) as ref_count
RETURN c.title, ref_count
ORDER BY ref_count DESC

7. 最短路径
MATCH (l1:Law {name: '中华人民共和国招标投标法'}), (l2:Law {name: '中华人民共和国政府采购法'}),
p = shortestPath((l1)-[*..5]-(l2))
RETURN p

（显式匹配关系后计数）
MATCH (n)-[r]-()
WITH n, labels(n) AS node_labels, count(r) AS degree
RETURN node_labels,
       count(n) AS num_nodes,
       avg(degree) AS average_degree
ORDER BY average_degree DESC;
